**Part A: Update Layout.tsx to render Sidebar**

Replace `packages/react-app/src/components/Layout.tsx` content. Import `Sidebar` from `./Sidebar`. Render `<Sidebar />` as first element inside a Fragment:

```tsx
import { Outlet, useParams } from 'react-router'
import { Sidebar } from './Sidebar'

export function Layout() {
  const { category, topicId } = useParams()
  const routeDisplay = category && topicId ? `${category} / ${topicId}` : null

  return (
    <>
      <Sidebar />
      <div className="p-4">
        <div className="flex items-center gap-2 mb-4">
          <span className="text-xs font-semibold text-white bg-cyan-400 rounded px-2 py-1">
            React 19
          </span>
          {routeDisplay && (
            <span className="text-sm text-slate-500">{routeDisplay}</span>
          )}
        </div>
        <Outlet />
        {!routeDisplay && (
          <div className="text-sm text-slate-400">
            React 19 -- Waiting for topic selection
          </div>
        )}
      </div>
    </>
  )
}
```
